010 all classes with labels.rq

 
# List the URIs and labels (optional) of all resources used as classes 
#
# @title	List of classes with their labels
# @tag 		model
# @tag 		SPARQL 1.0
# @see 		
# @param	
# 
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
SELECT DISTINCT ?class ?label WHERE {
  [] a ?class
  OPTIONAL { ?class rdfs:label ?label }
} ORDER BY ?class ?label

020 all classes with count.rq

 
# List the number of instances of each class, from largest to smallest
#
# @title	Number of instances per class
# @tag 		model
# @tag		SPARQL 1.0
# @see 		
# @param	
# 
SELECT DISTINCT ?class (COUNT(?x) AS ?numberOfInstances) WHERE {
  ?x a ?class .
}
GROUP BY ?class
ORDER BY DESC(?numberOfInstances)
